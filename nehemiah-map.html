<!DOCTYPE html>

<html>

<head>

  <title>Map of Israel & Judah - Time of Nehemiah</title>

  <meta charset="utf-8" />

  <meta name="viewport" content="width=device-width, initial-scale=1.0">

  <link rel="stylesheet" href="https://unpkg.com/leaflet/dist/leaflet.css" />

  <style>

    #map { height: 100vh; }

  </style>

</head>

<body>

<div id="map"></div>


<!-- Leaflet and Plugins -->

<script src="https://unpkg.com/leaflet/dist/leaflet.js"></script>

<script src="https://unpkg.com/leaflet-easyprint/dist/bundle.js"></script>


<script>

  var map = L.map('map').setView([31.5, 35.0], 8);


  L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

    attribution: 'Map data © OpenStreetMap contributors'

  }).addTo(map);


  // Village markers

  var locations = [

    { name: "Kirjath-Arba (Hebron)", coords: [31.5326, 35.0954] },

    { name: "Jeshua", coords: [31.681, 35.116] },

    { name: "Moladah", coords: [31.1833, 35.0333] },

    { name: "Beth Pelet", coords: [31.35, 34.7167] },

    { name: "Beersheba", coords: [31.2520, 34.7915] },

    { name: "Ziklag", coords: [31.35, 34.8833] },

    { name: "Lachish", coords: [31.5612, 34.8356] },

    { name: "Gittaim", coords: [31.8, 34.9833] },

    { name: "Hadid", coords: [32.0328, 34.9336] },

    { name: "Zeboiim", coords: [31.817, 35.05] },

    { name: "Neballat", coords: [31.933, 35.183] },

    { name: "Lod", coords: [31.9515, 34.8953] },

    { name: "Ono", coords: [32.025, 34.85] },

    { name: "Dibon", coords: [31.5, 35.3] }

  ];


  locations.forEach(function(loc) {

    L.marker(loc.coords).addTo(map)

      .bindPopup('<strong>' + loc.name + '</strong>');

  });


  // GeoJSON tribal boundaries

  var judah = {

    "type": "Feature",

    "properties": { "name": "Tribe of Judah" },

    "geometry": {

      "type": "Polygon",

      "coordinates": [[

        [34.7, 31.0], [34.8, 31.6], [35.1, 31.9], [35.2, 31.6],

        [35.3, 31.3], [35.1, 31.1], [34.9, 30.9], [34.7, 31.0]

      ]]

    }

  };


  var israel = {

    "type": "Feature",

    "properties": { "name": "Kingdom of Israel" },

    "geometry": {

      "type": "Polygon",

      "coordinates": [[

        [34.9, 32.0], [35.0, 32.8], [35.4, 33.2], [35.7, 32.8],

        [35.6, 32.1], [35.3, 31.9], [35.0, 31.8], [34.9, 32.0]

      ]]

    }

  };


  var judahLayer = L.geoJSON(judah, {

    style: {

      color: "darkred",

      fillColor: "#f76b6b",

      fillOpacity: 0.3,

      weight: 2

    }

  }).bindPopup("Tribe of Judah");


  var israelLayer = L.geoJSON(israel, {

    style: {

      color: "darkblue",

      fillColor: "#6ba6f7",

      fillOpacity: 0.3,

      weight: 2

    }

  }).bindPopup("Kingdom of Israel");


  judahLayer.addTo(map);

  israelLayer.addTo(map);


  L.control.layers(null, {

    "Tribe of Judah": judahLayer,

    "Kingdom of Israel": israelLayer

  }, { collapsed: false }).addTo(map);


  // EasyPrint control

  L.easyPrint({

    title: 'Export to PNG',

    position: 'topright',

    filename: 'nehemiah_map',

    exportOnly: true,

    hideControlContainer: false

  }).addTo(map);

</script>

</body>

</html>
